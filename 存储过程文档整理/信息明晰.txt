select YR,SUM(MONEY_ADD) MONEY_ADD from (
select convert(varchar(4),cus_pay.PAY_TIME,120) YR,cus_pay.MONEY_ADD
from MO_CON_PRO_INFO pro
join MO_CONTRACT_INFO con on pro.CONTRACT_ID = con.CONTRACT_ID
join V_SEARCH_SUB_PAY_NEW cus_pay on pro.CON_PRO_ID = cus_pay.CON_PRO_ID
join MO_CUSTOMER_INFO cus on con.C_ID = cus.C_ID
where 
cus_pay.PAY_TYPE <> '4' 
and cus_pay.PAY_TIME >='2004-1-1' and cus_pay.PAY_TIME <'2015-11-1' ) temp
group by temp.YR
order by temp.YR

------计算合同每年摊销
  ------计算摊销和剩余

  select CONTRACT_CODE,
	EveryMONEY*FOUR_YEAR as FOUR_YEAR_MONEY,
	case when FOUR_YEAR = 0 then 0
		 when FOUR_YEAR <> 0 then MONEY_PAY - EveryMONEY*(FOUR_YEAR + BeforeFOUR_YEAR) end as FOUR_YEAR_SHENGYU,
	EveryMONEY*FIVE_YEAR as FIVE_YEAR_MONEY,
	case when FIVE_YEAR = 0 then 0
		 when FIVE_YEAR <> 0 then MONEY_PAY - EveryMONEY*(FOUR_YEAR + FIVE_YEAR + BeforeFOUR_YEAR) end as FIVE_YEAR_SHENGYU,
	EveryMONEY*SIX_YEAR as SIX_YEAR_MONEY,
	case when SIX_YEAR = 0 then 0
		 when SIX_YEAR <> 0 then MONEY_PAY - EveryMONEY*( SIX_YEAR +FOUR_YEAR + FIVE_YEAR + BeforeFOUR_YEAR) end as SIX_YEAR_SHENGYU,
	EveryMONEY*SEVEN_YEAR as SEVEN_YEAR_MONEY,
	case when SEVEN_YEAR = 0 then 0
		 when SEVEN_YEAR <> 0 then MONEY_PAY - EveryMONEY*( SEVEN_YEAR+SIX_YEAR +FOUR_YEAR + FIVE_YEAR + BeforeFOUR_YEAR) end as SEVEN_YEAR_SHENGYU,
	EveryMONEY*EIGHT_YEAR as EIGHT_YEAR_MONEY,
	case when EIGHT_YEAR = 0 then 0
		 when EIGHT_YEAR <> 0 then MONEY_PAY - EveryMONEY*( EIGHT_YEAR+SEVEN_YEAR+SIX_YEAR +FOUR_YEAR + FIVE_YEAR + BeforeFOUR_YEAR) end as EIGHT_YEAR_SHENGYU,
	EveryMONEY*NINE_YEAR as NINE_YEAR_MONEY,
	case when NINE_YEAR = 0 then 0
		 when NINE_YEAR <> 0 then MONEY_PAY - EveryMONEY*( NINE_YEAR+EIGHT_YEAR+SEVEN_YEAR+SIX_YEAR +FOUR_YEAR + FIVE_YEAR + BeforeFOUR_YEAR) end as NINE_YEAR_SHENGYU,
	EveryMONEY*TEN_YEAR as TEN_YEAR_MONEY,
	case when TEN_YEAR = 0 then 0
		 when TEN_YEAR <> 0 then MONEY_PAY - EveryMONEY*( TEN_YEAR+NINE_YEAR+EIGHT_YEAR+SEVEN_YEAR+SIX_YEAR +FOUR_YEAR + FIVE_YEAR + BeforeFOUR_YEAR) end as TEN_YEAR_SHENGYU,
	EveryMONEY*ELEVEN_YEAR as ELEVEN_YEAR_MONEY,
	case when ELEVEN_YEAR = 0 then 0
		 when ELEVEN_YEAR <> 0 then MONEY_PAY - EveryMONEY*( ELEVEN_YEAR+TEN_YEAR+NINE_YEAR+EIGHT_YEAR+SEVEN_YEAR+SIX_YEAR +FOUR_YEAR + FIVE_YEAR + BeforeFOUR_YEAR) end as ELEVEN_YEAR_SHENGYU,
	EveryMONEY*TWELVE_YEAR as TWELVE_YEAR_MONEY,
	case when TWELVE_YEAR = 0 then 0
		 when TWELVE_YEAR <> 0 then MONEY_PAY - EveryMONEY*( TWELVE_YEAR+ELEVEN_YEAR+TEN_YEAR+NINE_YEAR+EIGHT_YEAR+SEVEN_YEAR+SIX_YEAR +FOUR_YEAR + FIVE_YEAR + BeforeFOUR_YEAR) end as TWELVE_YEAR_SHENGYU,
	EveryMONEY*THIRTEEN_YEAR as THIRTEEN_YEAR_MONEY,
	case when THIRTEEN_YEAR = 0 then 0
		 when THIRTEEN_YEAR <> 0 then MONEY_PAY - EveryMONEY*( THIRTEEN_YEAR+TWELVE_YEAR+ELEVEN_YEAR+TEN_YEAR+NINE_YEAR+EIGHT_YEAR+SEVEN_YEAR+SIX_YEAR +FOUR_YEAR + FIVE_YEAR + BeforeFOUR_YEAR) end as THIRTEEN_YEAR_SHENGYU,
	EveryMONEY*FOURTEEN_YEAR as FOURTEEN_YEAR_MONEY,
	case when FOURTEEN_YEAR = 0 then 0
		 when FOURTEEN_YEAR <> 0 then MONEY_PAY - EveryMONEY*( FOURTEEN_YEAR+THIRTEEN_YEAR+TWELVE_YEAR+ELEVEN_YEAR+TEN_YEAR+NINE_YEAR+EIGHT_YEAR+SEVEN_YEAR+SIX_YEAR +FOUR_YEAR + FIVE_YEAR + BeforeFOUR_YEAR) end as FOURTEEN_YEAR_SHENGYU,
	EveryMONEY*FIFTEEN_YEAR as FIFTEEN_YEAR_MONEY,
	case when FIFTEEN_YEAR = 0 then 0
		 when FIFTEEN_YEAR <> 0 then MONEY_PAY - EveryMONEY*(FIFTEEN_YEAR+ FOURTEEN_YEAR+THIRTEEN_YEAR+TWELVE_YEAR+ELEVEN_YEAR+TEN_YEAR+NINE_YEAR+EIGHT_YEAR+SEVEN_YEAR+SIX_YEAR +FOUR_YEAR + FIVE_YEAR + BeforeFOUR_YEAR) end as FIFTEEN_YEAR_SHENGYU
